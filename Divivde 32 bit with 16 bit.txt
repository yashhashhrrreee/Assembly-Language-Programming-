14. Program to Divide a 32 bit number with 16 bit number.

DATA SEGMENT
	K DW 7777
	M DW 77
	P DW 77
	J DW ?
	RES DB 10 DUP ('$')
DATA ENDS
CODE SEGMENT
ASSUME DS:DATA,CS:CODE
START:
	MOV AX,DATA
	MOV DS,AX
	MOV AX,K
	IMUL M
	CALL CHECKCARRY
	MOV J,AX
	LEA SI,RES
	CALL HEX2DEC
	LEA DX,RES
	MOV AH,9
	INT 21H
	MOV AH,4CH
	INT 21H
CODE ENDS
	CHECKCARRY PROC NEAR
	IDIV P
	RET
	CHECKCARRY ENDP
	HEX2DEC PROC NEAR
	MOV CX,0
	MOV BX,10
LOOP1: MOV DX,0
	DIV BX
	ADD DL,30H
	PUSH DX
	INC CX
	CMP AX,9
	JG LOOP1
	ADD AL,30H
	MOV [SI],AL
LOOP2: POP AX
	INC SI
	MOV [SI],AL
	LOOP LOOP2
	RET
	HEX2DEC ENDP
END START
